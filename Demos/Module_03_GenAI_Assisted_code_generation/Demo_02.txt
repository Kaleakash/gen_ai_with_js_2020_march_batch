const express = require('express');
const app = express();

app.use(express.json());

app.get('/api/hello', (req, res) => {
    res.json({ message: "Hello, World!" });
});

app.post('/api/hello', (req, res) => {
    res.json({ message: "Hello, World!" });
});

const PORT = 3000;
app.listen(PORT, () => {
    console.log(`Server is running on http://localhost:${PORT}`);
});
